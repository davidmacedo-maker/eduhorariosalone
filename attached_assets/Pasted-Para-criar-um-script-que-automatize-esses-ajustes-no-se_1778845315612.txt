Considerando que você quer modificar um arquivo PDF existente, veja abaixo a lógica estruturada do comando e do código que você pode rodar no seu ambiente Replit:🛠️ Código Python para o ReplitVocê precisará instalar as bibliotecas necessárias. Em seguida, utilize o seguinte script para aplicar as correções solicitadas:Pythonimport os
from pypdf import PdfReader, PdfWriter
from reportlab.pdfgen import canvas
from reportlab.lib.pagesizes import letter

def ajustar_livro_de_ponto():
    # 1. Configuração dos caminhos
    pdf_original = "livro de ponto.pdf"
    pdf_modificado = "livro_de_ponto_ajustado.pdf"
    pdf_mascara = "mascara_ajustes.pdf"

    if not os.path.exists(pdf_original):
        print(f"Erro: O arquivo {pdf_original} não foi encontrado.")
        return

    # 2. Criar uma máscara (overlay) para cobrir e adicionar os novos elementos
    c = canvas.Canvas(pdf_mascara, pagesize=letter)
    
    # --- REMOVER DATA, HORA E DIZERES DO RODAPÉ ---
    # Desenhamos retângulos brancos (sem borda) sobre as áreas que devem ser apagadas
    c.setFillColorRGB(1, 1, 1) # Cor branca
    
    # Apagar Data/Hora do topo (ex: "15/05/2026, 08:33")
    c.rect(10, 750, 150, 30, fill=True, stroke=False) 
    
    # Apagar o rodapé com link do replit e observações duplicadas/truncadas
    c.rect(10, 10, 580, 80, fill=True, stroke=False) 

    # --- AJUSTAR CAMPO DE ASSINATURA ---
    # Redesenhar uma área limpa para a assinatura do Diretor
    c.setFillColorRGB(0, 0, 0) # Cor preta para o texto
    c.setFont("Helvetica", 10)
    c.drawString(30, 40, "Assinatura do Diretor: _____________________________________")
    c.drawString(400, 40, "Data: ____/____/_______")

    # --- COLOCAR A PALAVRA "DIAS" NA VERTICAL ---
    # Primeiro limpamos o local antigo da palavra "Dias"
    c.setFillColorRGB(1, 1, 1)
    c.rect(15, 680, 30, 20, fill=True, stroke=False) # Ajustar coordenadas conforme seu layout
    
    # Escrevendo "DIAS" verticalmente caractere por caractere
    c.setFillColorRGB(0, 0, 0)
    c.setFont("Helvetica-Bold", 9)
    x_pos = 22  # Posição X na tabela
    y_pos = 690 # Posição Y inicial na tabela
    
    c.drawString(x_pos, y_pos, "D")
    c.drawString(x_pos, y_pos - 10, "I")
    c.drawString(x_pos, y_pos - 20, "A")
    c.drawString(x_pos, y_pos - 30, "S")

    c.save()

    # 3. Combinar o PDF original com a máscara de ajustes
    reader = PdfReader(pdf_original)
    writer = PdfWriter()
    mask_reader = PdfReader(pdf_mascara)

    # Aplica a máscara na primeira página
    pagina_original = reader.pages[0]
    pagina_original.merge_page(mask_reader.pages[0])
    writer.add_page(pagina_original)

    # Salva o resultado
    with open(pdf_modificado, "wb") as f_out:
        writer.write(f_out)

    # Limpa o arquivo temporário de máscara
    if os.path.exists(pdf_mascara):
        os.remove(pdf_mascara)

    print("Layout ajustado com sucesso! Arquivo salvo como: 'livro_de_ponto_ajustado.pdf'")

if __name__ == "__main__":
    ajustar_livro_de_ponto()
